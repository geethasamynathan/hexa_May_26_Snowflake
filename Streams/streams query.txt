SELECT 
    CUSTOMER_ID,
    CUSTOMER_NAME,
    CITY,
    EMAIL,
    UPDATE_AT,
    METADATA$ACTION,
    METADATA$ISUPDATE
FROM STREAMS_DEMO_DB.PUBLIC.CUSTMOER_STREAM;
//consume the stream object in target TABLE


MERGE INTO STREAMS_DEMO_DB.PUBLIC.CUSTOMERS_TARGET T
USING (
    SELECT
        CUSTOMER_ID,
        CUSTOMER_NAME,
        CITY,
        EMAIL,
        UPDATE_AT
    FROM STREAMS_DEMO_DB.PUBLIC.CUSTMOER_STREAM
    WHERE METADATA$ACTION = 'INSERT'
) S
ON T.CUSTOMER_ID = S.CUSTOMER_ID

WHEN MATCHED THEN UPDATE SET
    T.CUSTOMER_NAME = S.CUSTOMER_NAME,
    T.CITY = S.CITY,
    T.EMAIL = S.EMAIL,
    T.UPDATE_AT = S.UPDATE_AT

WHEN NOT MATCHED THEN INSERT
(
    CUSTOMER_ID,
    CUSTOMER_NAME,
    CITY,
    EMAIL,
    UPDATE_AT
)
VALUES
(
    S.CUSTOMER_ID,
    S.CUSTOMER_NAME,
    S.CITY,
    S.EMAIL,
    S.UPDATE_AT
);
